# import required library "turtle". turtle is a lightweight library for coding simple games
import turtle

# create screen object from the imported turtle library
turtle_screen = turtle.Screen()
turtle_screen.title("Pong v.2 by Benjamin Appelberg")
turtle_screen.bgcolor("white")
turtle_screen.setup(width=800, height=601)

# create [left] paddle rectangle with desired attributes
left_paddle = turtle.Turtle()
left_paddle.speed(0)
left_paddle.shape("square")
left_paddle.color("blue")
left_paddle.shapesize(stretch_wid=6, stretch_len=2)
left_paddle.penup()
left_paddle.goto(-400, 0)

# create [right] paddle rectangle with desired attributes
right_paddle = turtle.Turtle()
right_paddle.speed(0)
right_paddle.shape("square")
right_paddle.color("red")
right_paddle.shapesize(stretch_wid=6, stretch_len=2)
right_paddle.penup()
right_paddle.goto(+400, 0)

# create match ball
pong = turtle.Turtle()
pong.speed(40)
pong.shape("circle")
pong.color("yellow")
pong.penup()
pong.goto(0, 0)
pong.dx = 5
pong.dy = -5

# score-counters for each respective player
left_player_score_number = 0
left_player_score_number = 0

# display scoreboard
sketch = turtle.Turtle()
sketch.speed(0)
sketch.color("black")
sketch.penup()
sketch.hideturtle()
sketch.goto(0, 255)
sketch.write("Blue: 0           Right: 0", align="center", font=("Arial", 24, "bold"))





# Implementing the functions for moving paddle vertically
def paddle_left_up():
    y = left_paddle.ycor()
    y += 20
    left_paddle.sety(y)


def paddle_left_down():
    y = left_paddle.ycor()
    y -= 20
    left_paddle.sety(y)


def paddle_right_up():
    y = left_paddle.ycor()
    y += 20
    left_paddle.sety(y)


def paddle_right_down():
    y = left_paddle.ycor()
    y -= 20
    left_paddle.sety(y)


# key-bindings
turtle_screen.listen()
turtle_screen.onkeypress(paddle_left_up, "r")
turtle_screen.onkeypress(paddle_left_down(), "c")
turtle_screen.onkeypress(paddle_right_up(), "a")
turtle_screen.onkeypress(paddle_right_down(), "d")

# while program is running, update the turtle_screen_object.
while True:
    turtle_screen.update()
    pong.setx(pong.xcor() + pong.dx)
    pong.sety(pong.ycor() + pong.dy)
